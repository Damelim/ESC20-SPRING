In [19]:
fig = plt.figure(figsize=(10,5))

x = np.linspace(0, 15, 1000)
rate = [1,2,3]
shape = [1,2,3]

ax1 = fig.add_subplot(121)
for i in range(3):
    rv = gamma(a = 3, scale=1/rate[i])
    ax1.plot(x, rv.pdf(x), label='shape=3, rate={0}'.format(rate[i]))
ax1.legend()
ax1.set_title('Gamma Distribution')
ax1.set_xlabel('X')
ax1.set_ylabel('pdf(X)')

ax2 = fig.add_subplot(122)
for i in range(3):
    rv = gamma(a = shape[i], scale=1/0.5)
    ax2.plot(x, rv.pdf(x), label='shape={0}, rate=0.5'.format(shape[i]))
ax2.legend()
ax2.set_title('Gamma Distribution')
ax2.set_xlabel('X')
ax2.set_ylabel('pdf(X)')

fig.tight_layout()
fig.savefig('fig.png')
